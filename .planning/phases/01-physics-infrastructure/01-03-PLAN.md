```python
"""Phase 1 rule-based planner.

Strategy: each of our planets fires at the nearest enemy/neutral target it can afford
to capture (target.ships + 1, with a minimum of 5 ships). Garrison floor and per-turn
time budget are enforced here, not in physics.

Requirements satisfied:
  PHYS-04 : Never reduce a planet's garrison below max(production*2, 5)
  INFRA-01: Lives in agent/planner.py, importable from main.py
  INFRA-03: 0.8 s wall-clock guard truncates evaluation and returns best-so-far
"""
from __future__ import annotations

import math
import time  # MUST be `import time` (not `from time import time`) so tests can monkeypatch agent.planner.time.time
from collections import defaultdict
from typing import Any

from agent.physics import aim
from agent.state import parse_observation


# PHYS-04 garrison floor — locked formula from REQUIREMENTS.md
MIN_GARRISON: int = 5

# INFRA-03 wall-clock budget. CLAUDE.md says actTimeout = 1.0s; we leave 0.2s headroom.
TIME_BUDGET_SECONDS: float = 0.8


def garrison(planet) -> int:
    """PHYS-04: minimum ships that must remain on a planet after dispatch.

    Returns max(planet.production * 2, MIN_GARRISON).
    `planet` is duck-typed — needs only a `.production` attribute.
    """
    return max(planet.production * 2, MIN_GARRISON)


def select_moves(obs: Any) -> list:
    """Return the move list for this turn.

    Each move is [planet_id, angle_radians, num_ships]. Phase 1 strategy:
    nearest unowned planet, garrison-clamped, sun-aware (sun handled in physics.aim).
    """
    t0 = time.time()
    state = parse_observation(obs)

    # Classify planets
    my_planets = [p for p in state.planets if p.owner == state.player]
    targets = [
        p for p in state.planets
        if p.owner != state.player and p.id not in state.comet_planet_ids
    ]

    if not my_planets or not targets:
        return []

    reserved: dict[int, int] = defaultdict(int)
    moves: list = []

    for mine in my_planets:
        # INFRA-03: stop dispatching once we exceed the budget — return what we have.
        if time.time() - t0 > TIME_BUDGET_SECONDS:
            break

        # PHYS-04: enforce garrison floor.
        sendable = mine.ships - garrison(mine) - reserved[mine.id]
        if sendable <= 0:
            continue

        # Find nearest target by squared distance (no sqrt needed for comparison).
        nearest = min(
            targets,
            key=lambda t: (t.x - mine.x) ** 2 + (t.y - mine.y) ** 2,
            default=None,
        )
        if nearest is None:
            continue

        ships_needed = max(nearest.ships + 1, MIN_GARRISON)
        if ships_needed > sendable:
            continue

        angle, eta = aim(mine.id, nearest.id, ships_needed, obs)
        if angle is None or eta is None:
            continue

        moves.append([int(mine.id), float(angle), int(ships_needed)])
        reserved[mine.id] += ships_needed

    return moves


def agent(obs: Any) -> list:
    """Kaggle entry-point alias. Equivalent to select_moves(obs)."""
    return select_moves(obs)
```
