"""Physics primitives for Orbit Wars.

Constants and pure-math functions used by both the rule-based planner and (future) RL agent.
Algorithms verified against 07-claude_code.py and tested in tests/test_physics.py.

Requirements satisfied:
  PHYS-02: path_hits_sun rejects fleet paths crossing the sun exclusion zone
  PHYS-03: orbital_position uses the cumulative formula theta(t) = theta_0 + omega * t per D-03
  D-04:    INTERCEPT_LIMIT = 200 (raised from 20)
"""
from __future__ import annotations

import math

# --- Physics constants (load-bearing — match CLAUDE.md §Physics Facts) ----------
CENTER_X: float = 50.0
CENTER_Y: float = 50.0
SUN_R: float = 10.0
SUN_SAFE: float = 0.5           # total sun-exclusion radius = SUN_R + SUN_SAFE = 10.5
MAX_SPEED: float = 6.0
INTERCEPT_LIMIT: int = 200      # D-04: raised from 20; max turns to search for intercept


# --- Pure math primitives -------------------------------------------------------

def fleet_speed(ships: int) -> float:
    """Speed of a fleet of `ships` ships.

    Formula (per CLAUDE.md): 1.0 + (MAX_SPEED - 1.0) * (log(ships)/log(1000))^1.5
    1-ship fleets travel at 1.0. 1000+ ship fleets at MAX_SPEED.
    """
    if ships <= 1:
        return 1.0
    ratio = math.log(max(ships, 1)) / math.log(1000.0)
    ratio = max(0.0, min(1.0, ratio))
    return 1.0 + (MAX_SPEED - 1.0) * (ratio ** 1.5)


def is_orbiting(planet) -> bool:
    """True iff the planet's surface stays inside the orbit boundary (radius 50 from center).

    Per CLAUDE.md: orbiting iff dist(planet, center) + planet.radius < 50.
    """
    d = math.sqrt((planet.x - CENTER_X) ** 2 + (planet.y - CENTER_Y) ** 2)
    return d + planet.radius < 50.0


def orbital_position(
    x: float,
    y: float,
    angular_velocity: float,
    t: int,
) -> tuple[float, float]:
    """Position of an orbiting planet at turn `t` from its current (x, y).

    PHYS-03 / D-03: cumulative formula. (x, y) MUST be the original planet
    position from the current observation — never a previously-predicted position.

        r  = sqrt((x - 50)^2 + (y - 50)^2)
        theta = atan2(y - 50, x - 50) + omega * t
        x' = 50 + r * cos(theta)
        y' = 50 + r * sin(theta)
    """
    r = math.sqrt((x - CENTER_X) ** 2 + (y - CENTER_Y) ** 2)
    angle = math.atan2(y - CENTER_Y, x - CENTER_X) + angular_velocity * t
    return CENTER_X + r * math.cos(angle), CENTER_Y + r * math.sin(angle)


def _pt_seg_dist(
    px: float, py: float,
    x1: float, y1: float,
    x2: float, y2: float,
) -> float:
    """Perpendicular distance from point (px, py) to segment (x1,y1)-(x2,y2).

    Handles the zero-length-segment edge case by falling back to point-to-point distance.
    """
    dx, dy = x2 - x1, y2 - y1
    lsq = dx * dx + dy * dy
    if lsq < 1e-9:
        return math.sqrt((px - x1) ** 2 + (py - y1) ** 2)
    t = max(0.0, min(1.0, ((px - x1) * dx + (py - y1) * dy) / lsq))
    fx, fy = x1 + t * dx, y1 + t * dy
    return math.sqrt((px - fx) ** 2 + (py - fy) ** 2)


def path_hits_sun(mine, aim_x: float, aim_y: float, target_radius: float = 0.0) -> bool:
    """PHYS-02: True if the straight-line fleet path from `mine`'s surface to (aim_x, aim_y)
    passes within SUN_R + SUN_SAFE (= 10.5) of the sun center (50, 50).

    `mine` is duck-typed: must have .x, .y, .radius. The path starts at the surface
    of `mine` (center + unit * mine.radius) and ends `target_radius` short of (aim_x, aim_y).
    """
    d = math.sqrt((mine.x - aim_x) ** 2 + (mine.y - aim_y) ** 2)
    if d < 1e-6:
        return False
    dx, dy = (aim_x - mine.x) / d, (aim_y - mine.y) / d
    lx = mine.x + dx * mine.radius
    ly = mine.y + dy * mine.radius
    travel = max(0.0, d - mine.radius - target_radius)
    ex = lx + dx * travel
    ey = ly + dy * travel
    return _pt_seg_dist(CENTER_X, CENTER_Y, lx, ly, ex, ey) < SUN_R + SUN_SAFE

```python


# --- Intercept geometry (PHYS-01 fix) ------------------------------------------

def intercept_from_spawn(
    mine_x: float, mine_y: float, mine_r: float,
    tgt_x: float, tgt_y: float, tgt_r: float,
    angular_velocity: float, ships: int,
    t_max: int = INTERCEPT_LIMIT,
) -> tuple[int, float, float, float] | None:
    """PHYS-01: iterative spawn-surface intercept solver.

    The 09-Perfect_Aiming baseline computes the launch angle from the mine planet's
    CENTER, but the fleet actually spawns from the SURFACE
    (center + unit(angle) * mine_r). For mine_r=2.5 at range=50, this center-vs-surface
    offset causes a ~0.89-turn miss against fast-moving orbiting targets.

    Algorithm (verified in RESEARCH.md PHYS-01 Deep Dive):
      Pass 1: rough intercept from mine center -> initial angle estimate.
      Passes 2-4: re-solve from actual spawn point (mine_center + unit(angle) * mine_r).
                  Early-exit when |delta_angle| < 1e-6 between iterations.
                  After <=3 passes the residual miss drops from 0.89 turns to ~0.24.

    Returns (eta, angle_from_mine_CENTER, spawn_x, spawn_y) or None if no intercept
    found within t_max turns. The returned angle is measured from the mine CENTER
    (not from the spawn point) because the game engine encodes move angles relative
    to planet center.
    """
    speed = fleet_speed(ships)

    # Pass 1: rough intercept from mine CENTER
    angle: float | None = None
    for t in range(1, t_max + 1):
        px, py = orbital_position(tgt_x, tgt_y, angular_velocity, t)
        if math.sqrt((mine_x - px) ** 2 + (mine_y - py) ** 2) <= speed * t + tgt_r:
            angle = math.atan2(py - mine_y, px - mine_x)
            break
    if angle is None:
        return None

    # Passes 2-4: refine from actual spawn surface
    sx = mine_x + math.cos(angle) * mine_r
    sy = mine_y + math.sin(angle) * mine_r
    eta = -1
    for _ in range(3):
        found = False
        for t in range(1, t_max + 1):
            px, py = orbital_position(tgt_x, tgt_y, angular_velocity, t)
            if math.sqrt((sx - px) ** 2 + (sy - py) ** 2) <= speed * t + tgt_r:
                new_angle = math.atan2(py - mine_y, px - mine_x)  # angle from CENTER
                eta = t
                if abs(new_angle - angle) < 1e-6:
                    return eta, angle, sx, sy
                angle = new_angle
                sx = mine_x + math.cos(angle) * mine_r
                sy = mine_y + math.sin(angle) * mine_r
                found = True
                break
        if not found:
            return None
    if eta < 1:
        return None
    return eta, angle, sx, sy


# --- aim(): top-level physics interface used by planner AND future RL agent ----

def aim(
    origin_id: int,
    dest_id: int,
    fleet_size: int,
    obs,
) -> tuple[float, int] | tuple[None, None]:
    """Compute launch angle and ETA for a fleet from `origin_id` to `dest_id`.

    Returns:
      (angle_radians, eta_turns) — both finite, eta >= 1; OR
      (None, None) — only for geometric impossibility:
                       * no intercept found within INTERCEPT_LIMIT (=200) turns
                       * straight-line path crosses the sun exclusion zone
                       * origin_id or dest_id not present in obs.planets

    Does NOT enforce any garrison / ship-budget rule. Garrison lives in planner.py.

    Used by:
      * agent/planner.py (rule-based Phase 1 strategy)
      * future RL agent — the PPO action space picks (origin_id, dest_id) and aim()
        returns the angle, so physics correctness here directly unblocks the RL work.
    """
    # Parse the obs minimally — keep aim() decoupled from state.py.
    if isinstance(obs, dict):
        raw_planets = obs.get("planets", [])
        angular_velocity = float(obs.get("angular_velocity", 0.0))
    else:
        raw_planets = getattr(obs, "planets", [])
        angular_velocity = float(getattr(obs, "angular_velocity", 0.0))

    class _P:
        __slots__ = ("id", "owner", "x", "y", "radius", "ships", "production")

        def __init__(self, row):
            self.id = int(row[0])
            self.owner = int(row[1])
            self.x = float(row[2])
            self.y = float(row[3])
            self.radius = float(row[4])
            self.ships = int(row[5])
            self.production = int(row[6])

    origin = None
    dest = None
    for row in raw_planets:
        p = _P(row)
        if p.id == origin_id:
            origin = p
        if p.id == dest_id:
            dest = p
    if origin is None or dest is None:
        return None, None

    if is_orbiting(dest):
        result = intercept_from_spawn(
            origin.x, origin.y, origin.radius,
            dest.x, dest.y, dest.radius,
            angular_velocity, fleet_size,
            t_max=INTERCEPT_LIMIT,
        )
        if result is None:
            return None, None
        eta, angle, sx, sy = result
        # Sun check uses the predicted intercept point.
        ipx, ipy = orbital_position(dest.x, dest.y, angular_velocity, eta)
        if path_hits_sun(origin, ipx, ipy, target_radius=dest.radius):
            return None, None
        return angle, eta

    # Static target — straight shot from origin center to dest center.
    if path_hits_sun(origin, dest.x, dest.y, target_radius=dest.radius):
        return None, None
    dx = dest.x - origin.x
    dy = dest.y - origin.y
    distance = math.sqrt(dx * dx + dy * dy)
    if distance < 1e-9:
        return None, None
    speed = fleet_speed(fleet_size)
    eta = max(1, math.ceil(distance / speed))
    if eta > INTERCEPT_LIMIT:
        return None, None
    angle = math.atan2(dy, dx)
    return angle, int(eta)
```
