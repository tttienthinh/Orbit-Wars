```python
"""GameState dataclasses and observation parser for Orbit Wars."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any


@dataclass(slots=True)
class PlanetState:
    id: int
    owner: int
    x: float
    y: float
    radius: float
    ships: int
    production: int


@dataclass(slots=True)
class FleetState:
    id: int
    owner: int
    x: float
    y: float
    angle: float
    from_planet_id: int
    ships: int


@dataclass(slots=True)
class GameState:
    step: int
    player: int
    planets: list[PlanetState]
    fleets: list[FleetState]
    angular_velocity: float
    comet_planet_ids: frozenset[int] = field(default_factory=frozenset)


def parse_observation(obs: Any) -> GameState:
    """Parse a kaggle-environments observation (dict OR Namespace) into a GameState.

    obs["planets"][i] is a 7-tuple/list: (id, owner, x, y, radius, ships, production)
    obs["fleets"][i]  is a 7-tuple/list: (id, owner, x, y, angle, from_planet_id, ships)
    """

    def obs_get(key: str, default: Any) -> Any:
        if isinstance(obs, dict):
            return obs.get(key, default)
        return getattr(obs, key, default)

    planets = [
        PlanetState(
            id=int(row[0]),
            owner=int(row[1]),
            x=float(row[2]),
            y=float(row[3]),
            radius=float(row[4]),
            ships=int(row[5]),
            production=int(row[6]),
        )
        for row in obs_get("planets", [])
    ]
    fleets = [
        FleetState(
            id=int(row[0]),
            owner=int(row[1]),
            x=float(row[2]),
            y=float(row[3]),
            angle=float(row[4]),
            from_planet_id=int(row[5]),
            ships=int(row[6]),
        )
        for row in obs_get("fleets", [])
    ]
    return GameState(
        step=int(obs_get("step", 0)),
        player=int(obs_get("player", 0)),
        planets=planets,
        fleets=fleets,
        angular_velocity=float(obs_get("angular_velocity", 0.0)),
        comet_planet_ids=frozenset(int(i) for i in obs_get("comet_planet_ids", [])),
    )
```

# --- PHYS-03: cumulative orbital formula -----------------------------------------

def test_orbital_position_no_drift():
    """orbital_position must match the cumulative formula θ(t)=θ₀+ω·t exactly (no iterative drift)."""
    x0, y0, av = 60.0, 50.0, 0.05
    for t in [1, 10, 30, 60, 100, 200]:
        px, py = orbital_position(x0, y0, av, t)
        r = math.sqrt((x0 - CENTER_X) ** 2 + (y0 - CENTER_Y) ** 2)
        ang = math.atan2(y0 - CENTER_Y, x0 - CENTER_X) + av * t
        expected_x = CENTER_X + r * math.cos(ang)
        expected_y = CENTER_Y + r * math.sin(ang)
        assert abs(px - expected_x) < 1e-9, f"x mismatch at t={t}"
        assert abs(py - expected_y) < 1e-9, f"y mismatch at t={t}"


def test_orbital_position_t0_returns_input():
